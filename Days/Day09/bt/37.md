# 5. Sudoku Solver
Idea: Find empty cell. Try -> 1 to 9. If valid -> place, explore, undo


```
fun valid(r,c,val)
    for i in range(9)
        if board[r][i] == val
            return false
        if board[i][c] == val
            return false
    sr=(r/3)*3
    sc=(c/3)*3
    for i in range(sr,sr+3)
        for j in range(sc,sc+3)
            if board[i][j]==val
                return false
    return true
fun bt(board)
    for r <- 0..8
        for c <- 0..8
            if board[r][c]=='.'
                for val <- 1..9
                    if valid(r,c,val)
                        board[r][c]=val
                        if bt():
                            return true
                        board[r][c]='.'
                return false
fun solveSudoku(board)
    backtrack(board)
```
